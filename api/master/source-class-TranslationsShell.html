 55 /**
 56  * Create or update the file containing the translation strings for CRUD component usage
 57  *
 58  * @return void
 59  */
 60     public function generate() {
 61         if (!$this->_messages) {
 62             $this->_initializeMessages();
 63         }
 64 
 65         $models = $this->_getModels();
 66 
 67         if (!$models) {
 68             $this->out('<warning>No models found to be processed</warning>');
 69             return;
 70         }
 71 
 72         $this->hr();
 73         $this->out(sprintf('Generating translation strings for models: %s.', implode($models, ', ')));
 74         $this->out('');
 75 
 76         $path = $this->path();
 77 
 78         if (file_exists($path)) {
 79             $this->_strings = array_map('rtrim', file($path));
 80         } else {
 81             $this->_strings[] = '<?php';
 82         }
 83 
 84         $this->_generateTranslations(false);
 85         foreach ($models as $model) {
 86             $this->_generateTranslations($model);
 87         }
 88 
 89         return $this->_writeFile();
 90     }

 92 /**
 93  * _generateTranslations
 94  *
 95  * For the given model name, generate translation strings and add them to the _strings property
 96  * If the model name is false - the common messages are processed
 97  *
 98  * @param mixed $modelName name of a model or false for the general messages
 99  * @return void
100  */
101     protected function _generateTranslations($modelName) {
102         if ($modelName) {
103             $message = "$modelName CRUD Component translations";
104         } else {
105             $message = "Common CRUD Component translations";
106         }
107 
108         $this->_addDocBlock($message);
109         foreach ($this->_messages as $message) {
110             $type = strpos($message, '}') ? 'model' : 'common';
111 
112             if ($type === 'model' && $modelName || $type === 'common' && !$modelName) {
113                 $message = String::insert($message, array('name' => $modelName), array('before' => '{', 'after' => '}'));
114                 $string = "__d('crud', '$message');";
115 
116                 if (in_array($string, $this->_strings)) {
117                     $this->out('<info>Skipping:</info> ' . $message, 1, Shell::VERBOSE);
118                 } else {
119                     $this->out('<success>Adding:</success> ' . $message);
120                     $this->_strings[] = $string;
121                 }
122             }
123         }
124     }

126 /**
127  * _addDocBlock
128  *
129  * Add a doc block to the _strings property with the passed message appropriately formatted
130  * If the doc block already exists - return false
131  *
132  * @param string $message
133  * @return bool success
134  */
135     protected function _addDocBlock($message) {
136         $message = " * $message";
137 
138         if (in_array($message, $this->_strings)) {
139             return false;
140         }
141 
142         $this->_strings[] = '';
143         $this->_strings[] = '/**';
144         $this->_strings[] = $message;
145         $this->_strings[] = ' */';
146         return true;
147     }

149 /**
150  * _getModels
151  *
152  * If no arguments are passed to the cli call, return all App models
153  * Otherwise, assume the arguments are a list of file paths to plugin model dirs or an individual plugin model
154  *
155  * @return array
156  */
157     protected function _getModels() {
158         $objectType = 'Model';
159         $models = array();
160 
161         if ($this->args) {
162             foreach ($this->args as $arg) {
163                 preg_match('@Plugin/([^/]*)/?(?:Model/([^/]*))?@', $arg, $match);
164 
165                 if (!empty($match[2])) {
166                     $models[] = str_replace('.php', '', $match[2]);
167                 } elseif (!empty($match[1])) {
168                     $plugin = $match[1];
169                     $models = array_merge($models, App::objects("$plugin.Model"));
170                 }
171             }
172         } else {
173             $models = App::objects('Model');
174         }
175 
176         return $models;
177     }

197 /**
198  * _initializeMessages
199  *
200  * Extract all the messages used by the crud componentn, and write to the _messages property
201  *
202  * @return void
203  */
204     protected function _initializeMessages() {
205         $event = new TranslationsListener(new CrudSubject());
206         $defaults = $event->getDefaults();
207         foreach ($defaults as $key => $array) {
208             if (!is_array($array)) {
209                 continue;
210             }
211             foreach ($array as $subkey => $row) {
212                 if (!is_array($row) || !isset($row['message'])) {
213                     continue;
214                 }
215                 $this->_messages["$key.$subkey"] = $row['message'];
216             }
217         }
218     }
