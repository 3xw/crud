 84  * - init : Called before any other method in the decorator.

 86  * - recordNotFound : Called if a find() did not return any records
 87  * - beforePaginate : Called right before any paginate() method

134 /**
135  * Sets a configuration variable into this listener
136  *
137  * If called with no arguments, all configuration values are
138  * returned.
139  *
140  * $key is interpreted with dot notation, like the one used for
141  * Configure::write()
142  *
143  * If $key is string and $value is not passed, it will return the
144  * value associated with such key.
145  *
146  * If $key is an array and $value is empty, then $key will
147  * be interpreted as key => value dictionary of settings and
148  * it will be merged directly with $this->settings
149  *
150  * If $key is a string, the value will be inserted in the specified
151  * slot as indicated using the dot notation
152  *
153  * @param mixed $key
154  * @param mixed $value
155  * @return mixed|CrudAction
156  */
157     public function config($key = null, $value = null) {
158         if (is_null($key) && is_null($value)) {
159             return $this->_settings;
160         }
161 
162         if (is_null($value)) {
163             if (is_array($key)) {
164                 $this->_settings = $key + $this->_settings;
165                 return $this;
166             }
167 
168             return Hash::get($this->_settings, $key);
169         }
170 
171         if (is_array($value)) {
172             $value = $value + (array)Hash::get($this->_settings, $key);
173         }
174 
175         $this->_settings = Hash::insert($this->_settings, $key, $value);
176         return $this;
177     }
